微信小程序云开发

微信小程序云开发
by ZhangZiheng

云开发环境初始化

进行云开发之前，需要调用wx.cloud.init方法完成初始化。

// app.js
wx.cloud.init({
    env: 'env-id' // 云开发环境 id
})

云数据库的相关操作

增删改查基础操作：

  • 查询

    // 某个页面.js
onLoad() { // 小程序一进来时就会默认执行 onLoad 方法
    wx.cloud.database().collection('test-goods')
        .where({
        	name: 'apple' // 查询条件
 	    })
        .get()
        .then(res => { // 查询成功
        	console.log('success', res);
    	})
        .catch(err => { // 查询失败
        	console.log('fail', err);
    	})
}

  • 添加

    // 某个.js文件
wx.cloud.database().collection('test-goods')
    .add({ // 添加数据
        data: {
            name: '车厘子',
            price: 200
        }
	})
    .then(res => {
        console.log('添加成功', res);
    })
    .catch(err => {
        console.log('添加失败', err);
    })

  • 更新

    update() {
    wx.cloud.database().collection('test-goods')
      .where({
        name: '车厘子'
      })
      .update({ // 修改数据
        data: {
          price: 100
        }
      })
      .then(res => {
        console.log('修改成功', res);
      })
      .catch(err => {
        console.error('修改失败', err);
      })
 }

  • 删除

    remove() {
    wx.cloud.database().collection('test-goods')
      .where({
        name: '车厘子'
      })
      .remove() // 删除数据
      .then(res => {
        console.log('删除成功', res);
      })
      .catch(err => {
        console.error('删除失败', err);
      })
}

跳转详情页面：

  • 在wxml里用data-来定义需要传递的参数

    <!-- 在wxml里用 data- 来定义需要传递的参数 -->
<view wx:for="{{list}}">
  <view bindtap="goDetail" data-name="{{item.name}}">商品名: {{item.name}}, 价格: {{item.price}}</view>
</view>

  • 在 js 页面里的点击方法中获取需要的数据

    goDetail(event) {
    console.log('跳转', event);
    // 使用 wx.navigatorTo 进行跳转
    // ? 表示跳转时携带的参数
    wx.navigateTo({
        url: '/pages/goods-details/goods-details?name=' + event.currentTarget.dataset.name
    })
}

    接收时：

    onLoad(opt) { // 参数用来接收跳转时携带的参数
    console.log('列表携带的值', opt);
    var goods_name = opt.name; // 拿到参数
    wx.cloud.database().collection('test-goods')
        .where({
        	name: goods_name // 查找参数对应的物品
    	})
        .get()
        .then(res => {
        	console.log('商品详情页请求成功', res);
            this.setData({
                good: res.data
            })
   	 	})
        .catch(err => {
            console.error('商品详情页请求失败', err);
        })
}

获取用户输入数据并进行更新：

  • 输入：

    <!--pages/goods-items/goods-items.wxml-->
<text>pages/goods-items/goods-items.wxml</text>

<view>商品名:
  <input bindinput="getName" class="input"></input>
</view>

<view>商品价格:
  <input bindinput="getPrice" class="input"></input>
</view>
<button bindtap="addGood" type="primary">添加商品</button>

<!-- 用 data- 来传参 -->
<view wx:for="{{list}}">
  <view bindtap="goDetail" data-name="{{item.name}}">商品名: {{item.name}}, 价格: {{item.price}}</view>
</view>

  • 处理输入数据并进行更新

    // pages/goods-items/goods-items.js
let good_name  = '';
let good_price = '';
Page({
  onLoad() {
    this.getList();
  },

  getList() {
    wx.cloud.database().collection('test-goods')
      .get()
      .then(res => {
        console.log('商品列表请求成功', res);
        this.setData({
          list: res.data
        })
      })
      .catch(err => {
        console.error('商品列表请求失败', err);
      })
  },
  
  // 跳转商品详情页
  goDetail(event) {
    console.log('跳转', event);
    wx.navigateTo({
      url: '/pages/goods-details/goods-details?name=' + event.currentTarget.dataset.name
    })
  },

  // 获取用户输入的商品名
  getName(event) {
    good_name = event.detail.value;
    console.log('商品名称', good_name);
  },

  // 获取用户输入的商品价格
  getPrice(event) {
    good_price = event.detail.value;
    console.log('商品价格: ', good_price);
  },
  
  // 将商品添加到数据库
  addGood(event) {
    if (good_name == '') {
      wx.showToast({
        icon: 'none',
        title: '商品名不能为空',
      })
    }
    else if (good_price == '') {
      wx.showToast({
        icon: 'none',
        title: '商品价格不能为空',
      })
    }
    else {
      wx.cloud.database().collection('test-goods')
        .add({
          data: {
            name : good_name,
            price: good_price 
          }
        })
        .then(res => {
          console.log('添加成功', res);
          wx.showToast({
            title: '商品添加成功'
          })
          this.getList();
        })
        .catch(err => {
          console.log('添加失败', err);
        })
    }
  } 
})
